#!/usr/bin/python
# -*- coding: utf-8 -*-
from ege.models import Subject, Exam, ExamTest
from django.db.utils import IntegrityError

list_of_subjects = ["Математика", 'Информатика', 'Русский язык', 'Английский язык', 'Физика',
         'Химия', 'Биология', 'Обществознание', 'Литература', 'История']

exams = Exam.objects.all()
subjects = Subject.objects.all()

for exam in exams:
    for subj in list_of_subjects:
        try:
            subject = Subject(name=subj, type_of_exam=exam)
            subject.save()
            for i in range(1, 21):
                try:
                    tst = ExamTest(name='Вариант ' + str(i), subject=subject)
                    tst.save()
                except IntegrityError:
                    pass
        except IntegrityError:
            pass
